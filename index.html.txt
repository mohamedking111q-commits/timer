        <button class="break" onclick="setBreak()">Break 5 min</button>

    <audio id="alarmSound">
        <source src="https://www.soundjay.com/misc/sounds/bell-ringing-05.mp3">
    </audio>

    <script>
        let defaultTime = 1500;
        let breakTime = 300;
        let time = defaultTime;
        let interval;
        let isRunning = false;

        function updateTimer() {
            let minutes = Math.floor(time / 60);
            let seconds = time % 60;
            document.getElementById("timer").innerHTML =
                `${minutes}:${seconds < 10 ? '0' : ''}${seconds}`;
        }

        function startTimer() {
            if (isRunning) return;
            isRunning = true;

            document.getElementById("startBtn").classList.add("active-btn");

            interval = setInterval(() => {
                time--;
                updateTimer();

                if (time <= 0) {
                    document.getElementById("alarmSound").play();
                    clearInterval(interval);
                    isRunning = false;
                    document.getElementById("startBtn").classList.remove("active-btn");
                }
            }, 1000);
        }

        function pauseTimer() {
            clearInterval(interval);
            isRunning = false;
            document.getElementById("startBtn").classList.remove("active-btn");
        }

        function resetTimer() {
            pauseTimer();
            time = defaultTime;
            updateTimer();
        }

        function setBreak() {
            pauseTimer();
            time = breakTime;
            updateTimer();
        }
    </script>
